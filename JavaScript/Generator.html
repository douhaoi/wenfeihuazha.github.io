          # 概念

          # 基本使用

          function * foo(){
  console.log('start');

  try {
    const res = yield 'foo';
    console.log(res)
  } catch (error) {
    console.log(error)
  }
}
const generator = foo();

const result =  generator.next();
console.log(result);

generator.next('bar');

generator.throw(new Error('Generator error'))

          function ajax(){
  return Promise()

function * main(){
  const users = yield ajax('xxx');
  console.log(users)

const g = main();

const result = g.next();

result.value.then(data => {
  const result2 = g.next(data);

  result2.value.then(data => {
    g.next(data)
  })
